'use client';

import { useState, useEffect } from 'react';
import { supabase, BBSPost } from '@/lib/supabase';
import { User } from '@/lib/types';
import { canPost, getUserBadge } from '@/lib/auth-utils';
import PostCard from './PostCard';
import { MessageSquare, Send, LogIn, UserCircle } from 'lucide-react';

interface BBSProps {
  currentUser: User | null;
  onSignInClick: () => void;
}

export default function BBS({ currentUser, onSignInClick }: BBSProps) {
  const [posts, setPosts] = useState<BBSPost[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);
  const [title, setTitle] = useState('');
  const [content, setContent] = useState('');
  const [category, setCategory] = useState('general');
  const [userName, setUserName] = useState('');
  const [isSubmitting, setIsSubmitting] = useState(false);

  useEffect(() => {
    fetchPosts();
    
    const channel = supabase
      .channel('bbs_posts_changes')
      .on(
        'postgres_changes',
        {
          event: '*',
          schema: 'public',
          table: 'bbs_posts'
        },
        (payload) => {
          console.log('Real-time update:', payload);
          fetchPosts();
        }
      )
      .subscribe();

    return () => {
      supabase.removeChannel(channel);
    };
  }, []);

  const fetchPosts = async () => {
    try {
      setError(null);
      const { data, error: fetchError } = await supabase
        .from('bbs_posts')
        .select('*')
        .eq('is_active', true)
        .order('created_at', { ascending: false });

      if (fetchError) {
        console.error('Fetch error:', fetchError);
        setError(`Failed to load posts: ${fetchError.message}`);
        return;
      }

      setPosts(data || []);
    } catch (err) {
      console.error('Unexpected error:', err);
      setError('An unexpected error occurred while loading posts');
    } finally {
      setLoading(false);
    }
  };

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    
    if (!userName.trim()) {
      alert('Please enter your name');
      return;
    }

    if (!title.trim() || !content.trim()) {
      alert('Please fill in all fields');
      return;
    }

    setIsSubmitting(true);
    setError(null);

    try {
      const { data, error: insertError } = await supabase
        .from('bbs_posts')
        .insert([
          {
            title: title.trim(),
            content: content.trim(),
            category,
            user_name: userName.trim(),
            user_id: 'anonymous',
            likes: 0,
            is_active: true
          }
        ])
        .select();

      if (insertError) {
        console.error('Insert error:', insertError);
        setError(`Failed to create post: ${insertError.message}`);
        return;
      }

      setTitle('');
      setContent('');
      setCategory('general');
      
      console.log('Post created successfully:', data);
    } catch (err) {
      console.error('Unexpected error:', err);
      setError('An unexpected error occurred while creating the post');
    } finally {
      setIsSubmitting(false);
    }
  };

  if (loading) {
    return (
      <div className="flex flex-col items-center justify-center py-20">
        <MessageSquare className="w-16 h-16 text-gabriola-green animate-pulse mb-4" />
        <p className="text-xl text-gray-600">Loading community posts...</p>
      </div>
    );
  }

  return (
    <div className="max-w-4xl mx-auto px-4 py-12">
      <div className="text-center mb-12">
        <div className="flex items-center justify-center gap-4 mb-4">
          <h1 className="text-5xl font-bold text-gabriola-green">
            Community Bulletin Board
          </h1>
          {!currentUser && (
            <button
              onClick={onSignInClick}
              className="bg-white text-gabriola-green border-2 border-gabriola-green px-4 py-2 rounded-lg font-semibold hover:bg-gabriola-green hover:text-white transition-colors flex items-center gap-2"
            >
              <LogIn className="w-4 h-4" />
              Sign In to Post
            </button>
          )}
        </div>
        <p className="text-xl text-gray-600">
          Share news, ask questions, connect with neighbors
        </p>
        {currentUser && (
          <div className="mt-4 flex items-center justify-center gap-2 text-gray-600">
            <UserCircle className="w-5 h-5" />
            <span>Posting as <strong>{currentUser.username}</strong></span>
            {getUserBadge(currentUser) && (
              <span className={`${getUserBadge(currentUser)!.color} text-white px-2 py-0.5 rounded-full text-xs font-medium inline-flex items-center gap-1 ml-2`}>
                <span>{getUserBadge(currentUser)!.icon}</span>
                <span>{getUserBadge(currentUser)!.text}</span>
              </span>
            )}
          </div>
        )}
      </div>

      {error && (
        <div className="bg-red-50 border border-red-200 rounded-lg p-4 mb-8">
          <p className="text-red-800">{error}</p>
          <button
            onClick={() => {
              setError(null);
              fetchPosts();
            }}
            className="mt-2 text-sm text-red-600 hover:text-red-800 underline"
          >
            Try again
          </button>
        </div>
      )}

      {currentUser && canPost(currentUser) ? (
        <form onSubmit={handleSubmit} className="bg-white p-8 rounded-2xl shadow-lg mb-12 border-2 border-gabriola-green/20">
          <div className="flex items-center gap-3 mb-6">
            <Send className="w-6 h-6 text-gabriola-green" />
            <h2 className="text-2xl font-bold text-gray-800">Create a New Post</h2>
          </div>

          <div className="space-y-4">
            <div>
              <label htmlFor="userName" className="block text-sm font-medium text-gray-700 mb-1">
                Your Name
              </label>
              <input
                id="userName"
                type="text"
                value={userName}
                onChange={(e) => setUserName(e.target.value)}
                placeholder="Enter your name"
                required
                className="w-full p-4 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-gabriola-green focus:border-transparent"
              />
            </div>

            <div>
              <label htmlFor="category" className="block text-sm font-medium text-gray-700 mb-1">
                Category
              </label>
              <select
                id="category"
                value={category}
                onChange={(e) => setCategory(e.target.value)}
                className="w-full p-4 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-gabriola-green focus:border-transparent bg-white"
              >
                <option value="general">General Discussion</option>
                <option value="events">Community Events</option>
                <option value="buy-sell">Buy & Sell</option>
                <option value="lost-found">Lost & Found</option>
                <option value="recommendations">Recommendations</option>
                <option value="announcements">Announcements</option>
              </select>
            </div>

            <div>
              <label htmlFor="title" className="block text-sm font-medium text-gray-700 mb-1">
                Title
              </label>
              <input
                id="title"
                type="text"
                value={title}
                onChange={(e) => setTitle(e.target.value)}
                placeholder="Brief, descriptive title"
                required
                className="w-full p-4 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-gabriola-green focus:border-transparent"
              />
            </div>

            <div>
              <label htmlFor="content" className="block text-sm font-medium text-gray-700 mb-1">
                Message
              </label>
              <textarea
                id="content"
                value={content}
                onChange={(e) => setContent(e.target.value)}
                placeholder="What's on your mind?"
                required
                rows={6}
                className="w-full p-4 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-gabriola-green focus:border-transparent resize-none"
              />
            </div>

            <button
              type="submit"
              disabled={isSubmitting}
              className="w-full bg-gabriola-green text-white py-4 rounded-lg hover:bg-gabriola-green-dark transition-colors text-xl font-bold disabled:bg-gray-400 disabled:cursor-not-allowed flex items-center justify-center gap-2"
            >
              {isSubmitting ? (
                <>
                  <span className="animate-spin">‚è≥</span>
                  Posting...
                </>
              ) : (
                <>
                  <Send className="w-5 h-5" />
                  Post to the Island
                </>
              )}
            </button>
          </div>
        </form>
      ) : currentUser && !canPost(currentUser) ? (
        <div className="bg-yellow-50 border border-yellow-200 rounded-lg p-6 mb-8 text-center">
          <p className="text-yellow-800 text-lg">
            Your posting privileges have been temporarily restricted. Please contact a moderator if you believe this is an error.
          </p>
        </div>
      ) : null}

      <div className="space-y-6">
        {posts.length === 0 ? (
          <div className="text-center py-20 bg-white rounded-2xl shadow-md">
            <MessageSquare className="w-20 h-20 text-gray-300 mx-auto mb-4" />
            <p className="text-2xl text-gray-500 font-medium mb-2">
              No posts yet
            </p>
            <p className="text-gray-400">
              Be the first to start a conversation!
            </p>
          </div>
        ) : (
          <>
            <div className="text-center mb-6">
              <p className="text-gray-600">
                <strong>{posts.length}</strong> {posts.length === 1 ? 'post' : 'posts'} from the community
              </p>
            </div>
            {posts.map((post) => (
              <PostCard key={post.id} post={post} onRefresh={fetchPosts} />
            ))}
          </>
        )}
      </div>
    </div>
  );
}
