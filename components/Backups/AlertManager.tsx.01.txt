'use client';

import { useState } from 'react';
import { AlertTriangle, X, Plus, Clock } from 'lucide-react';

export interface Alert {
  id: string;
  title: string;
  message: string;
  severity: 'critical' | 'important' | 'advisory';
  issuer: string;
  organization: string;
  createdAt: Date;
  expiresAt?: Date;
  isActive: boolean;
}

interface AlertManagerProps {
  alerts: Alert[];
  onCreateAlert: (alert: Omit<Alert, 'id' | 'createdAt' | 'isActive'>) => void;
  onDismissAlert: (id: string) => void;
  onDeleteAlert: (id: string) => void;
}

export default function AlertManager({ 
  alerts, 
  onCreateAlert, 
  onDismissAlert,
  onDeleteAlert 
}: AlertManagerProps) {
  const [showCreateForm, setShowCreateForm] = useState(false);
  const [formData, setFormData] = useState({
    title: '',
    message: '',
    severity: 'important' as Alert['severity'],
    issuer: '',
    organization: '',
    expiresInHours: 24,
  });

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    
    const expiresAt = new Date();
    expiresAt.setHours(expiresAt.getHours() + formData.expiresInHours);

    onCreateAlert({
      title: formData.title,
      message: formData.message,
      severity: formData.severity,
      issuer: formData.issuer,
      organization: formData.organization,
      expiresAt,
    });

    // Reset form
    setFormData({
      title: '',
      message: '',
      severity: 'important',
      issuer: '',
      organization: '',
      expiresInHours: 24,
    });
    setShowCreateForm(false);
  };

  const getSeverityColor = (severity: Alert['severity']) => {
    switch (severity) {
      case 'critical':
        return 'from-red-600 to-red-700 border-red-800';
      case 'important':
        return 'from-orange-500 to-orange-600 border-orange-700';
      case 'advisory':
        return 'from-yellow-500 to-yellow-600 border-yellow-700';
    }
  };

  const getSeverityBadge = (severity: Alert['severity']) => {
    switch (severity) {
      case 'critical':
        return 'bg-red-600 text-white';
      case 'important':
        return 'bg-orange-500 text-white';
      case 'advisory':
        return 'bg-yellow-500 text-white';
    }
  };

  // Sort alerts by date (newest first)
  const sortedAlerts = [...alerts].sort((a, b) => 
    b.createdAt.getTime() - a.createdAt.getTime()
  );

  return (
    <div className="flex flex-col h-full bg-red-50">
      {/* DEMO MODE Banner */}
      <div className="bg-gradient-to-r from-purple-600 to-purple-700 text-white p-4 text-center">
        <p className="font-bold text-lg">ðŸŽ­ DEMO MODE - Mock Alerts for Testing Only</p>
        <p className="text-sm opacity-90 mt-1">
          These are sample alerts. Real alert system requires admin authentication.
        </p>
      </div>

      {/* Header */}
      <div className="p-4 bg-white border-b-2 border-red-200">
        <div className="flex items-center justify-between mb-3">
          <div className="flex items-center gap-3">
            <AlertTriangle className="w-8 h-8 text-red-600" />
            <div>
              <h1 className="text-2xl font-display font-bold text-red-600">
                Emergency Alerts
              </h1>
              <p className="text-sm text-gray-600">
                {sortedAlerts.length} active alert{sortedAlerts.length !== 1 ? 's' : ''}
              </p>
            </div>
          </div>
          <button
            onClick={() => setShowCreateForm(!showCreateForm)}
            className="flex items-center gap-2 bg-red-600 text-white px-4 py-2 rounded-lg font-semibold hover:bg-red-700 transition-colors"
          >
            {showCreateForm ? <X className="w-5 h-5" /> : <Plus className="w-5 h-5" />}
            {showCreateForm ? 'Cancel' : 'Create Alert'}
          </button>
        </div>
      </div>

      {/* Create Alert Form */}
      {showCreateForm && (
        <div className="p-4 bg-white border-b border-gray-200">
          <form onSubmit={handleSubmit} className="space-y-3">
            <div className="grid grid-cols-1 md:grid-cols-2 gap-3">
              <div>
                <label className="block text-sm font-medium text-gray-700 mb-1">
                  Alert Title *
                </label>
                <input
                  type="text"
                  value={formData.title}
                  onChange={(e) => setFormData({ ...formData, title: e.target.value })}
                  placeholder="e.g., Wildfire Warning"
                  required
                  className="w-full px-3 py-2 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-red-500"
                />
              </div>

              <div>
                <label className="block text-sm font-medium text-gray-700 mb-1">
                  Severity *
                </label>
                <select
                  value={formData.severity}
                  onChange={(e) => setFormData({ ...formData, severity: e.target.value as Alert['severity'] })}
                  className="w-full px-3 py-2 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-red-500"
                >
                  <option value="critical">Critical (Red)</option>
                  <option value="important">Important (Orange)</option>
                  <option value="advisory">Advisory (Yellow)</option>
                </select>
              </div>
            </div>

            <div>
              <label className="block text-sm font-medium text-gray-700 mb-1">
                Message *
              </label>
              <textarea
                value={formData.message}
                onChange={(e) => setFormData({ ...formData, message: e.target.value })}
                placeholder="Detailed alert message..."
                required
                rows={3}
                className="w-full px-3 py-2 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-red-500 resize-none"
              />
            </div>

            <div className="grid grid-cols-1 md:grid-cols-3 gap-3">
              <div>
                <label className="block text-sm font-medium text-gray-700 mb-1">
                  Your Name *
                </label>
                <input
                  type="text"
                  value={formData.issuer}
                  onChange={(e) => setFormData({ ...formData, issuer: e.target.value })}
                  placeholder="John Smith"
                  required
                  className="w-full px-3 py-2 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-red-500"
                />
              </div>

              <div>
                <label className="block text-sm font-medium text-gray-700 mb-1">
                  Organization *
                </label>
                <select
                  value={formData.organization}
                  onChange={(e) => setFormData({ ...formData, organization: e.target.value })}
                  required
                  className="w-full px-3 py-2 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-red-500"
                >
                  <option value="">Select...</option>
                  <option value="Fire Department">Fire Department</option>
                  <option value="RCMP">RCMP</option>
                  <option value="BC Ambulance">BC Ambulance</option>
                  <option value="Emergency Coordinator">Emergency Coordinator</option>
                  <option value="BC Ferries">BC Ferries</option>
                  <option value="Island Health">Island Health</option>
                  <option value="BC Hydro">BC Hydro</option>
                  <option value="Regional District">Regional District</option>
                </select>
              </div>

              <div>
                <label className="block text-sm font-medium text-gray-700 mb-1">
                  Expires In (hours)
                </label>
                <input
                  type="number"
                  value={formData.expiresInHours}
                  onChange={(e) => setFormData({ ...formData, expiresInHours: parseInt(e.target.value) || 24 })}
                  min="1"
                  max="720"
                  className="w-full px-3 py-2 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-red-500"
                />
              </div>
            </div>

            <button
              type="submit"
              className="w-full bg-red-600 text-white py-3 px-4 rounded-lg font-semibold hover:bg-red-700 transition-colors"
            >
              Publish Alert
            </button>
          </form>
        </div>
      )}

      {/* Alerts List */}
      <div className="flex-1 overflow-y-auto p-4">
        {sortedAlerts.length === 0 ? (
          <div className="text-center py-12">
            <AlertTriangle className="w-16 h-16 text-gray-300 mx-auto mb-4" />
            <p className="text-gray-500 text-lg font-medium">No active alerts</p>
            <p className="text-gray-400 text-sm mt-2">
              When emergencies occur, they will appear here
            </p>
          </div>
        ) : (
          <div className="space-y-4">
            {sortedAlerts.map((alert) => (
              <div
                key={alert.id}
                className={`bg-gradient-to-br ${getSeverityColor(alert.severity)} border-2 rounded-xl shadow-lg overflow-hidden`}
              >
                <div className="p-5 text-white">
                  {/* Header */}
                  <div className="flex items-start justify-between mb-3">
                    <div className="flex-1">
                      <div className="flex items-center gap-2 mb-2">
                        <span className={`px-3 py-1 rounded-full text-xs font-bold uppercase ${getSeverityBadge(alert.severity)}`}>
                          {alert.severity}
                        </span>
                        <span className="text-xs opacity-90">
                          {alert.organization}
                        </span>
                      </div>
                      <h3 className="text-xl font-bold mb-1">{alert.title}</h3>
                      <p className="text-sm opacity-90">Issued by: {alert.issuer}</p>
                    </div>
                    <div className="flex gap-2">
                      <button
                        onClick={() => onDismissAlert(alert.id)}
                        className="p-2 bg-white/20 hover:bg-white/30 rounded-lg transition-colors"
                        title="Dismiss"
                      >
                        <X className="w-5 h-5" />
                      </button>
                    </div>
                  </div>

                  {/* Message */}
                  <p className="text-white/95 leading-relaxed mb-4">
                    {alert.message}
                  </p>

                  {/* Footer */}
                  <div className="flex items-center justify-between text-xs opacity-90 pt-3 border-t border-white/20">
                    <div className="flex items-center gap-2">
                      <Clock className="w-4 h-4" />
                      <span>
                        Posted: {alert.createdAt.toLocaleString('en-US', { 
                          month: 'short', 
                          day: 'numeric', 
                          hour: 'numeric', 
                          minute: '2-digit' 
                        })}
                      </span>
                    </div>
                    {alert.expiresAt && (
                      <span>
                        Expires: {alert.expiresAt.toLocaleString('en-US', { 
                          month: 'short', 
                          day: 'numeric', 
                          hour: 'numeric', 
                          minute: '2-digit' 
                        })}
                      </span>
                    )}
                  </div>
                </div>

                {/* Delete button (admin only in real system) */}
                <div className="bg-black/20 px-5 py-2 flex justify-end">
                  <button
                    onClick={() => onDeleteAlert(alert.id)}
                    className="text-white text-xs hover:underline"
                  >
                    Delete Alert
                  </button>
                </div>
              </div>
            ))}
          </div>
        )}
      </div>
    </div>
  );
}
